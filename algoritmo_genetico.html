y_i = (1:5).^2;

cadenas = {['000111'; '010100'; '010110'; '110011'], ...
['010010'; '001100'; '101100'; '100110'], ...
['010101'; '101010'; '001101'; '101000'], ...
['100100'; '001001'; '101100'; '100011']}; % se registran los cromosomas iniciales
[x,y,S_x,L_x,S_y,VL_y,subtract] = get_MF(cadenas,y_i); % obtención de funciones de membresía
% promedio de la operación: 1000 - suma del error cuadrático
avg_e = mean(subtract);
% Adaptación relativa de los cromosomas
adapt_rel = subtract/avg_e;
% cromosoma más óptimo
[~,optimo] = max(adapt_rel);
figure(1) % entrada
plot(x,S_x(1,:),'r',x, L_x(1,:),'b', LineWidth=1.5)

ylabel('u_x')

xlabel('y')
ylabel('u_y')
title('Primer cromosoma')
legend('S_y','VL_y')
% gráfica de las funciones de membresía para el cromosoma más óptimo
figure(1) % entrada
plot(x,S_x(optimo,:),'r',x, L_x(optimo,:),'b', LineWidth=1.5)
xlabel('x')
ylabel('u_x')
title('Primera generación')
legend('S_x','L_x')
figure(2) % salida

xlabel('y')
ylabel('u_y')
title('Primera generación')
legend('S_y','VL_y')
% Primera generación

cadenas_optim = {[cadenas{1}],[cadenas{3}],[cadenas{3}],[cadenas{4}]};

for i = 1:4

for j = 1:length(cromosomas{1})
cromosomas_num(i,j) = str2double(cromosomas{i}(j)); % cromosomas puestos en vector como números
end
end
crossoverIndex_1 = 10; % Punto de cruce para las 2 primeras cadenas
crossoverIndex_2 = 20; % Punto de cruce para las 2 últimas cadenas
% Crossover
n_cadenas(1,:) = strrep(num2str([cromosomas_num(1,1:crossoverIndex_1) cromosomas_num(2,crossoverIndex_1+1:end)]),' ','');
n_cadenas(2,:) = strrep(num2str([cromosomas_num(2,1:crossoverIndex_1) cromosomas_num(1,crossoverIndex_1+1:end)]),' ','');
n_cadenas(3,:) = strrep(num2str([cromosomas_num(3,1:crossoverIndex_2) cromosomas_num(4,crossoverIndex_2+1:end)]),' ','');
n_cadenas(4,:) = strrep(num2str([cromosomas_num(4,1:crossoverIndex_2) cromosomas_num(3,crossoverIndex_2+1:end)]),' ','');
% Nuevas cadenas después del crossover de las más óptimas
new_cadenas_1 = {[n_cadenas(1,1:6); n_cadenas(1,7:12); n_cadenas(1,13:18); n_cadenas(1,19:24)],...

% se repite el proceso
[x,y,S_x,L_x,S_y,VL_y,subtract] = get_MF(new_cadenas_1,y_i); % obtención de funciones de membresía de la nueva generación

Función del algoritmo genético
function [x,y,S_x,L_x,S_y,VL_y,subtract] = get_MF(cadenas,y_i,optima)
% Recibe una variable tipo cell {}, en donde cada celda contiene un
% cromosoma, y recibe los datos del problema y_i.
dec = {[],[],[],[]};
for i = 1:4
for j = 1:4
dec{i}(j) = bin2dec(cadenas{i}(j,:)); % decodificación de las cadenas
end
end
% Convierte el número a un valor decimal entre el intervalo [Cmin,Cmax] para cada
% variable: "x", y "y".
% Inicialización de variables
n = 1000;
S_x = zeros(4,n); % MF S para "x"
L_x = zeros(4,n); % MF L para "x"
S_y = zeros(4,n); % MF S para "y"
VL_y = zeros(4,n); % MF VL para "y"
% Intervalo de la variable x
c_min_x = 0;
c_max_x = 5;
% variable x
x = linspace(c_min_x,c_max_x,n);
% Intervalo de la variable y
c_min_y = 0;
c_max_y = 25;
% variable y
y = linspace(c_min_y,c_max_y,n);
L = length(cadenas{1}(1,:)); % longitud de cada cadena por función
% Conversión de escala y creación de funciones de membresías
c = {[],[],[],[]};
for i = 1:4
for j = 1:4
if j < 3
c{i}(j) = c_min_x + (dec{i}(j)/(2^L-1))*(c_max_x-c_min_x); % valores en "x" establecidos en la escala [c_min_x, c_max_x]
else
c{i}(j) = c_min_y + (dec{i}(j)/(2^L-1))*(c_max_y-c_min_y); % valores en "x" establecidos en la escala [c_min_y, c_max_y]
end
end
% Variable x
S_x(i,:) = trimf(x, [c_min_x, c_min_x, c{i}(1)]); % Función S para "x"
L_x(i,:) = trimf(x,[c_max_x-c{i}(2), c_max_x, c_max_x ]); % Función L para "x"
% Variable y
S_y(i,:) = trimf(y, [c_min_y, c_min_y, c{i}(3)]); % Función S para "y"
VL_y(i,:) = trimf(y,[c_max_y-c{i}(4), c_max_y, c_max_y]); % Función VL para "y"
end
% Obtención de índices cuando x = 1,2,...,5
val_x = zeros(1,5);
for i = 1:5
val_x(i) = find(floor(x) == i,1);
end
% Reglas difusas
y_p = {[],[],[],[]};
for numero = 1:5
y_p = fuzzy_rules(y_p,S_x,L_x,S_y,VL_y,val_x,c_max_y,numero);
end
% Obtención de: 1000 - suma del error cuadrático
subtract = zeros(1,4);
for i = 1:4
subtract(i) = 1000-sum((y_i-y_p{i}).^2);
end
end
Reglas difusas
function [y_p] = fuzzy_rules(y_p,S_x,L_x,S_y,VL_y,val_x,c_max_y,num)
% Recibe funciones de membresía: S_x, L_x, S_y, VL_y
% Recibe valores donde la entrada "x" vale 1,2,...,5: val_x
% Recibe valor máximo de la salida: c_max_y
% Recibe el valor de "x" específico en el rango [0,5]
n = length(S_y);
for i = 1:4
if S_x(i,val_x(num)) ~= 0
y_p{i}(num) = find(floor(S_x(i,val_x(num))*100)/100 == floor(S_y(i,:)*100)/100,1)*c_max_y/n; % si "x" es S -> "y" es S
elseif L_x(i,val_x(num)) ~= 0
y_p{i}(num) = find(floor(L_x(i,val_x(num))*100)/100 == floor(VL_y(i,:)*100)/100,1)*c_max_y/n; % si "x" es L -> "y" es VL
else
y_p{i}(num) = 0;
end
end
end
